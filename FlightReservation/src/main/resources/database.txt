create database reservation


use reservation

CREATE TABLE `reservation`.`flight` (
  `ID` INT NOT NULL,
  `FLIGHT_NUMBER` VARCHAR(45) NOT NULL,
  `DEPARTMENT_CITY` VARCHAR(45) NOT NULL,
  `ARRIVEL_CITY` VARCHAR(45) NOT NULL,
  `DATE_OF_DEPARTURE` DATE NULL,
  `ESTIMATED_DEPARTURE_TIME` TIMESTAMP NULL,
  PRIMARY KEY (`ID`));


CREATE TABLE `reservation`.`passanger` (
  `ID` INT NOT NULL AUTO_INCREMENT,
  `FIRST_NAME` VARCHAR(45) NULL,
  `LAST_NAME` VARCHAR(45) NULL,
  `MIDDLE_NAME` VARCHAR(45) NULL,
  `EMAIL` VARCHAR(45) NULL,
  `PHONE` VARCHAR(10) NULL,
  PRIMARY KEY (`ID`));
